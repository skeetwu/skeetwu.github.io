for (ScenicAreaJson areaJson : LoaderWorker.scenicAreaList) {
      String cityname = areaJson.getCity();
      String province = areaJson.getProvince();
      String location = areaJson.getLocation();
      String scenicName = areaJson.getScenicName();
      String level = areaJson.getScenicLevel();
      if (!scenicName.contains("-sub"))
      {
        if ((cityname.indexOf(keyword) != -1) || (province.indexOf(keyword) != -1) || (location.indexOf(keyword) != -1) ||
          (scenicName.indexOf(keyword) != -1) || (level.indexOf(keyword) != -1)) {
          d_lat = areaJson.getLat().doubleValue();
          d_lng = areaJson.getLng().doubleValue();
          return computeDistance(Double.valueOf(d_lat), Double.valueOf(d_lng), limit, offset);
        }
      }
    }
